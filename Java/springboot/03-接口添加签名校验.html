接口添加签名校验

Joe's blog 2022-05-01 12:44:15 Spring SpringBoot

对于get接口使用签名校验可以增加接口被破解的难度，以下提供springboot拦截器校验以及js加签算法


package com.hozonauto.autodrive.hozondataplatform.interceptor;

import com.hozonauto.autodrive.hozondataplatform.constant.BootError;
import com.hozonauto.autodrive.hozondataplatform.exception.dto.BusinessException;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.codec.digest.DigestUtils;
import org.springframework.stereotype.Component;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.util.HashMap;
import java.util.Map;
import java.util.TreeMap;

@Component
@Slf4j
public class SignAuthInterceptor implements HandlerInterceptor {


    private static final String SECRET_KEY = "your_secret_key";

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {

        String sortedParams = sortAndConcatParams(getParamMap(request.getQueryString()));
        String serverSignature = generateSignature(sortedParams);
        String signature = request.getHeader("x-sign");
        // 验证签名是否匹配
        if (serverSignature.equals(signature)) {
            // 签名验证通过，处理请求
            return true;
        } else {
            // 签名验证失败，拒绝请求
            throw new BusinessException(BootError.BAD_SIGN_REQUEST);
        }
    }

    private Map<String, String> getParamMap(String queryString) {
        Map<String, String> paramMap = new HashMap<>();

        if (queryString != null && !queryString.isEmpty()) {
            String[] params = queryString.split("&");

            for (String param : params) {
                String[] keyValue = param.split("=");
                if (keyValue.length == 2) {
                    String key = keyValue[0];
                    String value = keyValue[1];
                    paramMap.put(key, value);
                }
            }
        }
        return paramMap;
    }

    private String sortAndConcatParams(Map<String, String> params) {
        TreeMap<String, String> sortedParams = new TreeMap<>(params);
        StringBuilder result = new StringBuilder();
        for (Map.Entry<String, String> entry : sortedParams.entrySet()) {
            result.append(entry.getKey()).append(entry.getValue());
        }
        return result.toString();
    }

    private String generateSignature(String input) {
        // 使用密钥对拼接后的字符串进行哈希运算
        return DigestUtils.sha256Hex(SECRET_KEY + input);
    }
}

# js 签名添加

postman脚本

// 替换为你的密钥
const SECRET_KEY = 'your_secret_key';
const urlParams = pm.request.url.query.members;
const paramsObject = urlParams.reduce((obj, param) => {
  obj[param.key] = param.value;
  return obj;
}, {});
const sortAndConcatParams = (params) =>
  Object.keys(params)
    .sort()
    .map((key) => key + params[key])
    .join('');
const sortedParams = sortAndConcatParams(paramsObject);

const signature = CryptoJS.SHA256(SECRET_KEY + sortedParams).toString(CryptoJS.enc.Hex);
console.log(signature)
// 添加签名到请求头
pm.request.headers.add({
    key: 'x-sign',
    value: signature
});
