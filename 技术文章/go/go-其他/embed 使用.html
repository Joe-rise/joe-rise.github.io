    # 需求描述

    项目打包启动后需要读取配置文件，不需要更改配置文件时可以将toml文件打包进项目中

    # 使用embed实现

    在config.go嵌入文件

    package config

import (
	"embed"
	"fmt"
	"github.com/spf13/viper"
)

//go:embed config.toml
var embedConfig embed.FS

var c TomlConfig

func init() {
	// 从嵌入的文件中读取配置
	data, err := embedConfig.ReadFile("config.toml")
	if err != nil {
		panic(fmt.Errorf("fatal error reading embedded config file: %s", err))
	}

	// 设置viper配置
	viper.SetConfigType("toml")
	if err := viper.ReadConfig(bytes.NewBuffer(data)); err != nil {
		panic(fmt.Errorf("fatal error config file: %s", err))
	}

	// 解析配置
	if err := viper.Unmarshal(&c); err != nil {
		panic(fmt.Errorf("fatal error unmarshaling config: %s", err))
	}
}

    # 优先读取外部配置文件

    package config

import (
	"embed"
	"fmt"
	"github.com/spf13/viper"
	"os"
)

//go:embed config.toml
var embedConfig embed.FS

var c TomlConfig

func init() {
	// 尝试从外部文件读取配置
	if err := readExternalConfig(); err != nil {
		// 如果外部文件不存在或出现其他错误，则从嵌入的文件中读取配置
		fmt.Println("Reading external config failed. Trying embedded config...")
		if err := readEmbeddedConfig(); err != nil {
			panic(fmt.Errorf("fatal error reading config: %s", err))
		}
	}
}

func readExternalConfig() error {
	// 检查外部文件是否存在
	externalConfigPath := "path/to/external/config.toml" // 替换为你的外部配置文件路径
	if _, err := os.Stat(externalConfigPath); os.IsNotExist(err) {
		// 外部文件不存在
		fmt.Printf("External config file not found at %s\n", externalConfigPath)
		return err
	}

	// 从外部文件读取配置
	viper.SetConfigFile(externalConfigPath)
	return viper.ReadInConfig()
}

func readEmbeddedConfig() error {
	// 从嵌入的文件中读取配置
	data, err := embedConfig.ReadFile("config.toml")
	if err != nil {
		return err
	}

	// 设置viper配置
	viper.SetConfigType("toml")
	if err := viper.ReadConfig(bytes.NewBuffer(data)); err != nil {
		return err
	}

	// 解析配置
	return viper.Unmarshal(&c)
}
